021    package cascading.tuple.hadoop.util;
022    
023    import java.io.IOException;
024    import java.util.Comparator;
025    
026    import cascading.CascadingException;
027    import cascading.tuple.io.TuplePair;
028    import org.apache.hadoop.conf.Configuration;
029    
030    /** Class TuplePairComparator is an implementation of {@link org.apache.hadoop.io.RawComparator}. */
031    public class GroupingSortingComparator extends DeserializerComparator<TuplePair>
032      {
033      Comparator[] sortComparators;
034    
035      @Override
036      public void setConf( Configuration conf )
037        {
038        super.setConf( conf );
039    
040        if( conf == null )
041          return;
042    
043        sortComparators = deserializeComparatorsFor( "cascading.sort.comparator" );
044        sortComparators = delegatingComparatorsFor( sortComparators );
045        }
046    
047      public int compare( byte[] b1, int s1, int l1, byte[] b2, int s2, int l2 )
048        {
049        try
050          {
051          lhsBuffer.reset( b1, s1, l1 );
052          rhsBuffer.reset( b2, s2, l2 );
053    
054          int c = compareTuples( groupComparators );
055    
056          if( c != 0 )
057            return c;
058    
059          return compareTuples( sortComparators );
060          }
061        catch( IOException exception )
062          {
063          throw new CascadingException( exception );
064          }
065        finally
066          {
067          lhsBuffer.clear();
068          rhsBuffer.clear();
069          }
070        }
071    
072      public int compare( TuplePair lhs, TuplePair rhs )
073        {
074        int c = compareTuples( groupComparators, lhs.getLhs(), rhs.getLhs() );
075    
076        if( c != 0 )
077          return c;
078    
079        return compareTuples( sortComparators, lhs.getRhs(), rhs.getRhs() );
080        }
081      }
