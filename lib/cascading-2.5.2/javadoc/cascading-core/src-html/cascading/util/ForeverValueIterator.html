021    package cascading.util;
022    
023    /** The ForeverValueIterator class will iterate over a single value forever. */
024    public class ForeverValueIterator<Value> extends SingleValueIterator<Value>
025      {
026      public ForeverValueIterator( Value value )
027        {
028        super( value );
029        }
030    
031      @Override
032      public Value next()
033        {
034        return value;
035        }
036      }
