043    /**
044     * Class Fields represents the field names in a {@link Tuple}. A tuple field may be a literal String value representing a
045     * name, or it may be a literal Integer value representing a position, where positions start at position 0.
046     * A Fields instance may also represent a set of field names and positions.
047     * <p/>
048     * Fields are used as both declarators and selectors. A declarator declares that a given {@link Tap} or
049     * {@link cascading.operation.Operation} returns the given field names, for a set of values the size of
050     * the given Fields instance. A selector is used to select given referenced fields from a Tuple.
051     * For example; <br/>
052     * <code>Fields fields = new Fields( "a", "b", "c" );</code><br/>
053     * This creates a new Fields instance with the field names "a", "b", and "c". This Fields instance can be used as both
054     * a declarator or a selector, depending on how it's used.
055     * <p/>
056     * Or For example; <br/>
057     * <code>Fields fields = new Fields( 1, 2, -1 );</code><br/>
058     * This creates a new Fields instance that can only be used as a selector. It would select the second, third, and last
059     * position from a given Tuple instance, assuming it has at least four positions. Since the original field names for those
060     * positions will carry over to the new selected Tuple instance, if the original Tuple only had three positions, the third
061     * and last positions would be the same, and would throw an error on there being duplicate field names in the selected
062     * Tuple instance.
063     * <p/>
064     * Additionally, there are eight predefined Fields sets used for different purposes; {@link #NONE}, {@link #ALL}, {@link #GROUP},
065     * {@link #VALUES}, {@link #ARGS}, {@link #RESULTS}, {@link #UNKNOWN}, {@link #REPLACE}, and {@link #SWAP}.
066     * <p/>
067     * The {@code NONE} Fields set represents no fields.
068     * <p/>
069     * The {@code ALL} Fields set is a "wildcard" that represents all the current available fields.
070     * <p/>
071     * The {@code GROUP} Fields set represents all the fields used as grouping values in a previous {@link cascading.pipe.Splice}.
072     * If there is no previous Group in the pipe assembly, the GROUP represents all the current field names.
073     * <p/>
074     * The {@code VALUES} Fields set represent all the fields not used as grouping fields in a previous Group.
075     * <p/>
076     * The {@code ARGS} Fields set is used to let a given Operation inherit the field names of its argument Tuple. This Fields set
077     * is a convenience and is typically used when the Pipe output selector is {@code RESULTS} or {@code REPLACE}.
078     * <p/>
079     * The {@code RESULTS} Fields set is used to represent the field names of the current Operations return values. This Fields
080     * set may only be used as an output selector on a Pipe. It effectively replaces in the input Tuple with the Operation result
081     * Tuple.
082     * <p/>
083     * The {@code UNKNOWN} Fields set is used when Fields must be declared, but how many and their names is unknown. This allows
084     * for arbitrarily length Tuples from an input source or some Operation. Use this Fields set with caution.
085     * <p/>
086     * The {@code REPLACE} Fields set is used as an output selector to inline replace values in the incoming Tuple with
087     * the results of an Operation. This is a convenience Fields set that allows subsequent Operations to 'step' on the
088     * value with a given field name. The current Operation must always use the exact same field names, or the {@code ARGS}
089     * Fields set.
090     * <p/>
091     * The {@code SWAP} Fields set is used as an output selector to swap out Operation arguments with its results. Neither
092     * the argument and result field names or size need to be the same. This is useful for when the Operation arguments are
093     * no longer necessary and the result Fields and values should be appended to the remainder of the input field names
094     * and Tuple.
095     */

214      /**
215       * Method join joins all given Fields instances into a new Fields instance.
216       * <p/>
217       * Use caution with this method, it does not assume the given Fields are either selectors or declarators. Numeric position fields are left untouched.
218       * <p/>
219       * If the resulting set of fields and ordinals is length zero, {@link Fields#NONE} will be returned.
220       *
221       * @param fields of type Fields
222       * @return Fields
223       */
224      public static Fields join( Fields... fields )
225        {
226        return join( false, fields );
227        }

229      public static Fields join( boolean maskDuplicateNames, Fields... fields )
230        {
231        int size = 0;
232    
233        for( Fields field : fields )
234          {
235          if( field.isSubstitution() || field.isUnknown() )
236            throw new TupleException( "cannot join fields if one is a substitution or is unknown" );
237    
238          size += field.size();
239          }
240    
241        if( size == 0 )
242          return Fields.NONE;
243    
244        Comparable[] elements = join( size, fields );
245    
246        if( maskDuplicateNames )
247          {
248          Set<String> names = new HashSet<String>();
249    
250          for( int i = elements.length - 1; i >= 0; i-- )
251            {
252            Comparable element = elements[ i ];
253    
254            if( names.contains( element ) )
255              elements[ i ] = i;
256            else if( element instanceof String )
257              names.add( (String) element );
258            }
259          }
260    
261        Type[] types = joinTypes( size, fields );
262    
263        if( types == null )
264          return new Fields( elements );
265        else
266          return new Fields( elements, types );
267        }
268    
269      private static Comparable[] join( int size, Fields... fields )
270        {
271        Comparable[] elements = expand( size, 0 );
272    
273        int pos = 0;
274        for( Fields field : fields )
275          {
276          System.arraycopy( field.fields, 0, elements, pos, field.size() );
277          pos += field.size();
278          }
279    
280        return elements;
281        }
282    
283      private static Type[] joinTypes( int size, Fields... fields )
284        {
285        Type[] elements = new Type[ size ];
286    
287        int pos = 0;
288        for( Fields field : fields )
289          {
290          if( field.isNone() )
291            continue;
292    
293          if( field.types == null )
294            return null;
295    
296          System.arraycopy( field.types, 0, elements, pos, field.size() );
297          pos += field.size();
298          }
299    
300        return elements;
301        }

303      public static Fields mask( Fields fields, Fields mask )
304        {
305        Comparable[] elements = expand( fields.size(), 0 );
306    
307        System.arraycopy( fields.fields, 0, elements, 0, elements.length );
308    
309        for( int i = elements.length - 1; i >= 0; i-- )
310          {
311          Comparable element = elements[ i ];
312    
313          if( element instanceof Integer )
314            continue;
315    
316          if( mask.getIndex().containsKey( element ) )
317            elements[ i ] = i;
318          }
319    
320        return new Fields( elements );
321        }

323      /**
324       * Method merge merges all given Fields instances into a new Fields instance where a merge is a set union of all the
325       * given Fields instances.
326       * <p/>
327       * Thus duplicate positions or field names are allowed, they are subsequently discarded in favor of the first
328       * occurrence. That is, merging "a" and "a" would yield "a", not "a, a", yet merging "a,b" and "c" would yield "a,b,c".
329       * <p/>
330       * Use caution with this method, it does not assume the given Fields are either selectors or declarators. Numeric position fields are left untouched.
331       *
332       * @param fields of type Fields
333       * @return Fields
334       */
335      public static Fields merge( Fields... fields )
336        {
337        List<Comparable> elements = new ArrayList<Comparable>();
338        List<Type> elementTypes = new ArrayList<Type>();
339    
340        for( Fields field : fields )
341          {
342          for( Comparable comparable : field )
343            {
344            if( !elements.contains( comparable ) )
345              {
346              elements.add( comparable );
347              elementTypes.add( field.getType( comparable ) );
348              }
349            }
350          }
351    
352        Comparable[] comparables = elements.toArray( new Comparable[ elements.size() ] );
353        Type[] types = elementTypes.toArray( new Type[ elementTypes.size() ] );
354    
355        if( Util.containsNull( types ) )
356          return new Fields( comparables );
357    
358        return new Fields( comparables, types );
359        }

361      public static Fields copyComparators( Fields toFields, Fields... fromFields )
362        {
363        for( Fields fromField : fromFields )
364          {
365          for( Comparable field : fromField )

368    
369            if( comparator != null )
370              toFields.setComparator( field, comparator );
371            }
372          }
373    
374        return toFields;
375        }

395      private static Comparable[] expand( int size, int startPos )
396        {
397        if( size < 1 )
398          throw new TupleException( "invalid size for fields: " + size );
399    
400        if( startPos < 0 )
401          throw new TupleException( "invalid start position for fields: " + startPos );
402    
403        Comparable[] fields = new Comparable[ size ];
404    
405        for( int i = 0; i < fields.length; i++ )
406          fields[ i ] = i + startPos;
407    
408        return fields;
409        }

411      /**
412       * Method resolve returns a new selector expanded on the given field declarations
413       *
414       * @param selector of type Fields
415       * @param fields   of type Fields
416       * @return Fields
417       */
418      public static Fields resolve( Fields selector, Fields... fields )
419        {
420        boolean hasUnknowns = false;
421        int size = 0;
422    
423        for( Fields field : fields )
424          {
425          if( field.isUnknown() )
426            hasUnknowns = true;
427    
428          if( !field.isDefined() && field.isUnOrdered() )
429            throw new TupleException( "unable to select from field set: " + field.printVerbose() );
430    
431          size += field.size();
432          }
433    
434        if( selector.isAll() )
435          {
436          Fields result = fields[ 0 ];
437    
438          for( int i = 1; i < fields.length; i++ )
439            result = result.append( fields[ i ] );
440    
441          return result;
442          }
443    
444        if( selector.isReplace() )
445          {
446          if( fields[ 1 ].isUnknown() )
447            throw new TupleException( "cannot replace fields with unknown field declaration" );
448    
449          if( !fields[ 0 ].contains( fields[ 1 ] ) )
450            throw new TupleException( "could not find all fields to be replaced, available: " + fields[ 0 ].printVerbose() + ",  declared: " + fields[ 1 ].printVerbose() );
451    
452          Type[] types = fields[ 0 ].getTypes();
453    
454          if( types != null )
455            {
456            for( int i = 1; i < fields.length; i++ )
457              {
458              Type[] fieldTypes = fields[ i ].getTypes();
459              if( fieldTypes == null )
460                continue;
461    
462              for( int j = 0; j < fieldTypes.length; j++ )
463                fields[ 0 ] = fields[ 0 ].applyType( fields[ i ].get( j ), fieldTypes[ j ] );
464              }
465            }
466    
467          return fields[ 0 ];
468          }
469    
470        // we can't deal with anything but ALL
471        if( !selector.isDefined() )
472          throw new TupleException( "unable to use given selector: " + selector );
473    
474        Set<String> notFound = new LinkedHashSet<String>();
475        Set<String> found = new HashSet<String>();
476        Fields result = size( selector.size() );
477    
478        if( hasUnknowns )
479          size = -1;
480    
481        Type[] types = null;
482    
483        if( size != -1 )
484          types = new Type[ size ];
485    
486        int offset = 0;
487        for( Fields current : fields )
488          {
489          if( current.isNone() )
490            continue;
491    
492          resolveInto( notFound, found, selector, current, result, types, offset, size );
493          offset += current.size();
494          }
495    
496        if( types != null && !Util.containsNull( types ) ) // don't apply types if any are null
497          result = result.applyTypes( types );
498    
499        notFound.removeAll( found );
500    
501        if( !notFound.isEmpty() )
502          throw new FieldsResolverException( new Fields( join( size, fields ) ), new Fields( notFound.toArray( new Comparable[ notFound.size() ] ) ) );
503    
504        if( hasUnknowns )
505          return selector;
506    
507        return result;
508        }
509    
510      private static void resolveInto( Set<String> notFound, Set<String> found, Fields selector, Fields current, Fields result, Type[] types, int offset, int size )
511        {
512        for( int i = 0; i < selector.size(); i++ )
513          {
514          Comparable field = selector.get( i );
515    
516          if( field instanceof String )
517            {
518            int index = current.indexOfSafe( field );
519    
520            if( index == -1 )
521              notFound.add( (String) field );
522            else
523              result.set( i, handleFound( found, field ) );
524    
525            if( index != -1 && types != null && current.getType( index ) != null )
526              types[ i ] = current.getType( index );
527    
528            continue;
529            }
530    
531          int pos = current.translatePos( (Integer) field, size ) - offset;
532    
533          if( pos >= current.size() || pos < 0 )
534            continue;
535    
536          Comparable thisField = current.get( pos );
537    
538          if( types != null && current.getType( pos ) != null )
539            types[ i ] = current.getType( pos );
540    
541          if( thisField instanceof String )
542            result.set( i, handleFound( found, thisField ) );
543          else
544            result.set( i, field );
545          }
546        }
547    
548      private static Comparable handleFound( Set<String> found, Comparable field )
549        {
550        if( found.contains( field ) )
551          throw new TupleException( "field name already exists: " + field );
552    
553        found.add( (String) field );
554    
555        return field;
556        }

558      /**
559       * Method asDeclaration returns a new Fields instance for use as a declarator based on the given fields value.
560       * <p/>
561       * Typically this is used to convert a selector to a declarator. Simply, all numeric position fields are replaced
562       * by their absolute position.
563       * <p/>
564       * Comparators are preserved in the result.
565       *
566       * @param fields of type Fields
567       * @return Fields
568       */
569      public static Fields asDeclaration( Fields fields )
570        {
571        if( fields == null )
572          return null;
573    
574        if( fields.isNone() )
575          return fields;
576    
577        if( !fields.isDefined() )
578          return UNKNOWN;
579    
580        if( fields.isOrdered() )
581          return fields;
582    
583        Fields result = size( fields.size() );
584    
585        copy( null, result, fields, 0 );
586    
587        result.types = copyTypes( fields.types, result.size() );
588        result.comparators = fields.comparators;
589    
590        return result;
591        }

593      private static Fields asSelector( Fields fields )
594        {
595        if( !fields.isDefined() )
596          return UNKNOWN;
597    
598        return fields;
599        }

814        {

917      final int[] getPos( Fields fields, int tupleSize )
918        {
919        // test for key, as we stuff a null value
920        if( !isUnknown() && getPosCache().containsKey( fields ) )
921          return getPosCache().get( fields );
922

930        if( size() == 0 && isUnknown() )
931          return translatePos( fields, tupleSize );

933        int[] pos = translatePos( fields, size() );

935        return putReturn( fields, pos );

1034      /**
1035       * Method select returns a new Fields instance with fields specified by the given selector.
1036       *
1037       * @param selector of type Fields
1038       * @return Fields
1039       */
1040      public Fields select( Fields selector )
1041        {
1042        if( !isOrdered() )
1043          throw new TupleException( "this fields instance can only be used as a selector" );
1044    
1045        if( selector.isAll() )
1046          return this;
1047    
1048        // supports -1_UNKNOWN_RETURNED
1049        // guarantees pos arguments remain selector positions, not absolute positions
1050        if( isUnknown() )
1051          return asSelector( selector );
1052    
1053        if( selector.isNone() )
1054          return NONE;
1055    
1056        Fields result = size( selector.size() );
1057    
1058        for( int i = 0; i < selector.size(); i++ )
1059          {
1060          Comparable field = selector.get( i );
1061    
1062          if( field instanceof String )
1063            {
1064            result.set( i, get( indexOf( field ) ) );
1065            continue;
1066            }
1067    
1068          int pos = translatePos( (Integer) field );
1069    
1070          if( this.get( pos ) instanceof String )
1071            result.set( i, this.get( pos ) );
1072          else
1073            result.set( i, pos ); // use absolute position if no field name
1074          }
1075    
1076        if( this.types != null )
1077          {
1078          result.types = new Type[ result.size() ];
1079    
1080          for( int i = 0; i < selector.size(); i++ )
1081            {
1082            Comparable field = selector.get( i );
1083    
1084            if( field instanceof String )
1085              result.setType( i, getType( indexOf( field ) ) );
1086            else
1087              result.setType( i, getType( translatePos( (Integer) field ) ) );
1088            }
1089          }
1090    
1091        return result;
1092        }

1216       * <p/>

1218       * <p/>
1219       * Note any relative positional elements are retained, thus appending two Fields each declaring {@code -1}
1220       * position will result in a TupleException noting duplicate fields.
1221       * <p/>
1222       * See {@link #subtract(Fields)} for removing field names.

1232      private Fields appendInternal( Fields fields, boolean isSelect )
1233        {
1234        if( fields == null )
1235          return this;
1236    
1237        // allow unordered fields to be appended to build more complex selectors
1238        if( this.isAll() || fields.isAll() )
1239          throw new TupleException( "cannot append fields: " + this.print() + " + " + fields.print() );
1240    
1241        if( ( this.isUnknown() || this.size() == 0 ) && fields.isUnknown() )
1242          return UNKNOWN;
1243    
1244        if( fields.isNone() )
1245          return this;
1246    
1247        if( this.isNone() )
1248          return fields;
1249    
1250        Set<Comparable> names = new HashSet<Comparable>();
1251    
1252        // init the Field
1253        Fields result = size( this.size() + fields.size() );
1254    
1255        // copy over field names from this side
1256        copyRetain( names, result, this, 0, isSelect );
1257        // copy over field names from that side
1258        copyRetain( names, result, fields, this.size(), isSelect );
1259    
1260        if( this.isUnknown() || fields.isUnknown() )
1261          result.kind = Kind.UNKNOWN;
1262    
1263        if( ( this.isNone() || this.types != null ) && fields.types != null )
1264          {
1265          result.types = new Type[ this.size() + fields.size() ];
1266    
1267          if( this.types != null ) // supports appending to NONE
1268            System.arraycopy( this.types, 0, result.types, 0, this.size() );
1269    
1270          System.arraycopy( fields.types, 0, result.types, this.size(), fields.size() );
1271          }
1272    
1273        return result;
1274        }

1277       * Method rename will rename the from fields to the values in to to fields. Fields may contain field names, or

1287        {

1289          throw new TupleException( "cannot rename fields in a substitution or unknown Fields instance: " + this.print() );
1290    
1291        if( from.size() != to.size() )
1292          throw new TupleException( "from and to fields must be the same size" );
1293

1528              {
1529              if( i + 1 == size() || !( get( i + 1 ) instanceof Number ) )
1530                {
1531                if( buffer.length() != 0 )
1532                  buffer.append( ", " );
1533    
1534                if( startIndex != i )
1535                  buffer.append( startIndex ).append( ":" ).append( field );
1536                else
1537                  buffer.append( i );
1538    
1539                startIndex = i;
1540                }
1541    
1542              continue;
1543              }
1544    
1545            if( i != 0 )
1546              buffer.append( ", " );
1547    
1548            if( field instanceof String )
1549              buffer.append( "\'" ).append( field ).append( "\'" );
1550            else if( field instanceof Fields )
1551              buffer.append( ( (Fields) field ).print() );
1552    
1553            startIndex = i + 1;
1554            }
1555          }
1556    
1557        if( kind != null )
1558          {
1559          if( buffer.length() != 0 )
1560            buffer.append( ", " );
1561          buffer.append( kind );
1562          }
1563    
1564        return buffer.toString();
1565        }
